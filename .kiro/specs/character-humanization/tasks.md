**Описание:**

**Код:**

```typescript
/**
 * Модуль адаптации стиля общения под пользователя
 */

export interface UserStyle {
  messageLength: 'short' | 'medium' | 'long'
  formality: 'casual' | 'neutral' | 'formal'
  emotionalTone: 'excited' | 'neutral' | 'frustrated' | 'curious'
  technicalLevel: 'beginner' | 'intermediate' | 'advanced'
  needsClarification: boolean
}

/**
 * Анализировать стиль сообщения пользователя
 */
export function analyzeUserStyle(message: string, history: Array<{ role: string; content: string }>): UserStyle {
  const lowerMessage = message.toLowerCase()
  const wordCount = message.split(/\s+/).length
  
  // Определяем длину сообщения
  let messageLength: UserStyle['messageLength'] = 'medium'
  if (wordCount < 10) messageLength = 'short'
  else if (wordCount > 50) messageLength = 'long'
  
  // Определяем формальность
  let formality: UserStyle['formality'] = 'neutral'
  const formalWords = ['пожалуйста', 'благодарю', 'уважаемый', 'прошу']
  const casualWords = ['привет', 'спасибо', 'круто', 'окей', 'ок']
  
  const hasFormal = formalWords.some(w => lowerMessage.includes(w))
  const hasCasual = casualWords.some(w => lowerMessage.includes(w))
  
  if (hasFormal && !hasCasual) formality = 'formal'
  else if (hasCasual && !hasFormal) formality = 'casual'
  
  // Определяем эмоциональный тон
  let emotionalTone: UserStyle['emotionalTone'] = 'neutral'
  
  const excitedWords = ['круто', 'вау', 'отлично', 'супер', 'интересно']
  const frustratedWords = ['не понимаю', 'не получается', 'опять', 'снова', 'ошибка', 'не работает']
  const curiousWords = ['почему', 'как', 'зачем', 'интересно', 'расскажи', 'объясни']
  
  if (excitedWords.some(w => lowerMessage.includes(w)) || message.includes('!')) {
    emotionalTone = 'excited'
  } else if (frustratedWords.some(w => lowerMessage.includes(w))) {
    emotionalTone = 'frustrated'
  } else if (curiousWords.some(w => lowerMessage.includes(w)) && message.includes('?')) {
    emotionalTone = 'curious'
  }
  
  // Определяем технический уровень
  let technicalLevel: UserStyle['technicalLevel'] = 'intermediate'
  
  const beginnerPhrases = ['что такое', 'как начать', 'с чего начать', 'не знаю', 'первый раз']
  const advancedPhrases = ['оптимизация', 'архитектура', 'паттерн', 'алгоритм', 'сложность']
  
  if (beginnerPhrases.some(p => lowerMessage.includes(p))) {
    technicalLevel = 'beginner'
  } else if (advancedPhrases.some(p => lowerMessage.includes(p))) {
    technicalLevel = 'advanced'
  }
  
  // Проверяем, нужно ли уточнение
  const needsClarification = shouldAskClarification(message)
  
  return {
    messageLength,
    formality,
    emotionalTone,
    technicalLevel,
    needsClarification
  }
}

/**
 * Проверить, нужно ли задать уточняющий вопрос
 */
export function shouldAskClarification(message: string): boolean {
  const lowerMessage = message.toLowerCase()
  
  // Слишком короткий и неоднозначный
  if (message.split(/\s+/).length < 5 && !message.includes('?')) {
    return false // Слишком короткий, но не вопрос
  }
  
  // Неоднозначные слова
  const ambiguousWords = [
    'это', 'оно', 'такое', 'подобное', 'похожее',
    'лучше', 'правильно', 'нормально'
  ]
  
  const hasAmbiguous = ambiguousWords.some(w => lowerMessage.includes(w))
  
  // Вопрос без контекста
  const isQuestionWithoutContext = 
    message.includes('?') && 
    message.split(/\s+/).length < 15 &&
    hasAmbiguous
  
  return isQuestionWithoutContext
}

/**
 * Получить инструкции для адаптации ответа
 */
export function getStyleAdaptationInstructions(userStyle: UserStyle): string {
  let instructions = '\n\nАДАПТАЦИЯ СТИЛЯ:\n'
  
  // Длина ответа
  if (userStyle.messageLength === 'short') {
    instructions += '- Отвечай КРАТКО (2-3 абзаца максимум)\n'
  } else if (userStyle.messageLength === 'long') {
    instructions += '- Можешь дать ПОДРОБНЫЙ ответ с примерами\n'
  }
  
  // Формальность
  if (userStyle.formality === 'casual') {
    instructions += '- Используй ДРУЖЕЛЮБНЫЙ тон, можно на "ты"\n'
  } else if (userStyle.formality === 'formal') {
    instructions += '- Используй АКАДЕМИЧЕСКИЙ стиль, на "вы"\n'
  }
  
  // Эмоциональный тон
  if (userStyle.emotionalTone === 'excited') {
    instructions += '- Поддержи ЭНТУЗИАЗМ, покажи увлечённость темой\n'
  } else if (userStyle.emotionalTone === 'frustrated') {
    instructions += '- Будь ТЕРПЕЛИВЫМ и поддерживающим, упрости объяснение\n'
  } else if (userStyle.emotionalTone === 'curious') {
    instructions += '- Поощри ЛЮБОПЫТСТВО, дай интересные детали\n'
  }
  
  // Технический уровень
  if (userStyle.technicalLevel === 'beginner') {
    instructions += '- Объясняй ПРОСТЫМИ словами, избегай жаргона\n'
  } else if (userStyle.technicalLevel === 'advanced') {
    instructions += '- Можешь использовать ТЕХНИЧЕСКИЕ термины\n'
  }
  
  // Уточнение
  if (userStyle.needsClarification) {
    instructions += '- ОБЯЗАТЕЛЬНО задай уточняющий вопрос перед ответом\n'
  }
  
  return instructions
}
```
